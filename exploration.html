Board Game Exploration

cThese are the ideas and the board game pieces. Our board game involved around warfare. Here is the steps: At beginning of game: Each player starts at one corner of the board; pieces of artillery placed randomly. Does not matter where you go; you just try to have your colored artillery on the board. During the Game: Roll the die. Pick whichever way you want to go. Whether you land on a square or star, you do something. Objective of Game: Be the last person standing; (your artillery is last to leave).

  from random import randint
WIDTH = 600
HEIGHT = 400
score = 0
game_over = False
hedgehog = Actor("hedgehog")
hedgehog.pos = 100, 100

coin = Actor("coin")
coin.pos = 200, 200

def draw():
    screen.blit('background', (0, 0))
    hedgehog.draw()
    coin.draw()
    screen.draw.text("Score: " + str(score), color="black", topleft=(10,10))

    if game_over:
        screen.fill("pink")
        screen.draw.text("Final Score: " + str(score), topleft=(10, 10), fontsize=60)

def place_coin():
    coin.x = randint(20, (WIDTH - 20))
    coin.y = randint(20, (HEIGHT - 20))

def time_up():
    global game_over
    game_over = True

def update():
    global score

    if keyboard.left:
        hedgehog.x = hedgehog.x - 7
    elif keyboard.right:
        hedgehog.x = hedgehog.x + 7
    elif keyboard.up:
        hedgehog.y = hedgehog.y - 7
    elif keyboard.down:
        hedgehog.y = hedgehog.y +


    coin_collected = hedgehog.colliderect(coin)


    if coin_collected:
        score = score + 10
        place_coin()


clock.schedule(time_up, 45.0)
place_coin()
